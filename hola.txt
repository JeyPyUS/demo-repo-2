def rellenar_fila(fila, np_curr, f_value, f_curr, df_input):
    '''
    Llenado de fila
    '''
    if fila != "cod_currency":
        np_fila = df_input[fila].to_numpy()
        np_fila[f_value[np.where(np.isnan(np_fila[f_value]))]] = -99999
        np_fila = rellenar(fila, np_fila)
        np_fila[np.where(np_fila==-99999)] = np.nan
        return np_fila
    else:
        np_curr[f_value] = f_curr
        np_curr = rellenar(fila, np_curr)
        return np_curr
