def execute_query(query: str, is_fw=False, conexion_ar=con_db_AR, conexion_fw=con_db_FW):
    """
    Ejecuta una query sobre una base de datos y su respectivo commit

    Parameters:
        query(str): Texto con la query a realizar
        is_fw(boolean): Si debe ejecutarse en las tablas del FW o en RedShift
        conexion_redshift: Conexion a RedShift
        conexion_fw: Conexion al framework
    """
    if is_fw:
        conexion = conexion_fw
    else:
        conexion = conexion_ar

    conexion.execute(f"{query}; commit;")
