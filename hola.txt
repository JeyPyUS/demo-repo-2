def API_post(url, json, api_key, obj_auth):
    dict_headers = {"VND.com.blackrock.Request-ID": str(uuid.uuid4()),
                    "VND.com.blackrock.Origin-Timestamp": str(
                    datetime.utcnow().replace(microsecond=0).astimezone().isoformat()),
                    "VND.com.blackrock.API-Key": api_key}

    ans = requests.post(url=url, data=json,  headers=dict_headers, auth=obj_auth)
    return ans.text
