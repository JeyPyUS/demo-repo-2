def write_log(log: str, id_ej: int) -> None:
    """
    Escritura en el fichero de log

    Parameters:
        log (str): Mensaje que debe ser escrito
        id_ej (int): Numero de la ejecucion actual
    """
    timestamp = datetime.now().strftime("%d-%m-%Y (%H:%M:%S)")
    full_path = sys.argv[0]
    path = full_path[:full_path.rfind("/")]
    with open(f'{path}/log_{id_ej}.txt', 'a') as f:
        f.write(f"[{timestamp}] - {log}\n")
