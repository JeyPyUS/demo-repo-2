def fill_bad_values(lista, past_value, df, f_a_calcular, dict_class):
    '''
    Rellenado de campos no v√°lidos
    '''
    dummy = np.zeros(len(df))
    dummy[f_a_calcular] = past_value

    for i,j in enumerate(lista):
        if len(j)!=0:
            dummy[j] = df.rent.values[dict_class[i][0]]

    past_value = dummy[f_a_calcular]
    return past_value
