def rellenar(fila, columna):
    '''
    Rellenar los valores con el valor anterior
    '''
    if fila != "cod_currency":
        mask = np.isnan(columna)
    else:
        mask = ~(columna==columna)
    idx = np.where(~mask,np.arange(mask.size),0)
    np.maximum.accumulate(idx, out=idx)
    columna = columna[idx]
    return columna
